Future<void> deleteWorkflow(String userId, String workflowId) async {

    Response response =
        await dio.delete(postWorkflowUrl(userId) + "/$workflowId");

    if (response.statusCode != 200) {
      throw Exception("Error deleting workflow");
    }
  } catch (e) {
    rethrow;
  }
}
